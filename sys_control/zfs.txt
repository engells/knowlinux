建立資料儲存池(zpool create pool)
    zpool create pool_name DeviceNode
        zpool create XXXX /dev/sda[#] # 建立名稱為 XXXX 的 pool，對應的設備代號為 /dev/zvol/XXXX
        zpool create -m /dir_name XXXX /dev/sda[#] # 在 /dev/sda[#] 設備上建立 XXXX pool，並掛載在 /dir_name wd1t 目錄
    zpool create pool_name mirror /dev/sda2 /dev/sda3
        # 建立 Raid1(1+1=1) Mirror 格式，同一資料同步寫入兩顆 HD，注意 mirror 參數
        # 另有 raidz1(>= 3 HDD) 及 raidz3 (>= 8 HDD?)
    zpool create -f -o ashift pool_name /dev/sdb /dev/sdc # 建立 Raid 0 需 1 顆或以上硬碟
    zfs set compression=lz4 pool_name # 啟動壓縮機制

    zpool list # 列出已經建立的 pool
    zpool status # 列出已經建立的 pool 的狀態

    zpool offline pool_name /dev/sda2 # 讓 pool_name 這個 pool 的 /dev/sda2 硬碟暫時離線
    zpool online pool_name /dev/sda2 # 讓 pool_name 這個 pool 的 /dev/sda2 硬碟上線

    zpool export -f pool_name # 卸除(umount) pool_name 這個 pool 參數 -f (Force) 為強制卸除
    zpool import pool_name # 掛載(mount) pool_name 這個 pool

    zpool scrub pool_name # 手動驗證或清潔 pool_name 這個 pool 中所有資料的完整性，取消指令為 zpool scrub -s pool_name
    zpool replace pool_name /dev/sda4 /dev/sda5 # 在 pool_name pool 中用新的 /dev/sda4 /dev/sda5 取代原本的 /dev/sda2 /dev/sda3 兩棵 HD

刪除資料儲存池
    zpool destroy pool_name

zfs create -V 8G pool_name/dataset_name # 在 storage 這個 pool 中建立 compressed 分割區/檔案目錄(DataSet)，容量為 8G
zfs destroy pool_name/dataset_name # 可刪除 pool_name/dataset_name 目錄(磁區)

zfs set compression=gzip pool_name/dataset_name # 將 dataset_name # 目錄(磁區)設定成 gzip 壓縮格式
zfs set compression=off pool_name/dataset_name # 解除 dataset_name # 解除目錄(磁區)設的壓縮格式

zfs mount # 查看所有 ZFS 掛載的磁區
zfs set mountpoint=/path/to/target pool_name/XXXX/YYYY
zfs mount pool_name/XXXX/YYYY
zfs get all pool_name/dataset_name # 查看 pool_name/dataset_name 檔案目錄屬性

zfs set quota=3G pool_name/dataset_name # 可設定檔案空間大小為 3G 但 pool_name 這個 pool 總大小為 5.8G 如果 3G 滿了而 5.8 還未用完會 share 給 /pool_name/dataset_name 繼續使用
zfs set reservation=3g pool_name/dataset_name # 設定保留了 3G 給 pool_name/dataset_name，pool_namee 的 Size 變成 2.8G(5.8-3.0)

zfs set mountpoint=/data storage/data # 掛載點由 /storage/data 變成 /data
zfs set sharenfs=rw storage/data # 設定分享權限可讀及寫，通常用在 NFS 分享上

touch /pool_name/dataset_name/1234 # 建一個檔案到 compressed 目錄下
md5 /pool_name/dataset_name # 用 md5 去 hash snapshot 之前的目錄，會得到一個 hash 值：MD5 (/pool_name/dataset_name) = 7bffed2808dfba7915f89f8f42b09f83

zfs snapshot pool_name/dataset_name@2009-04-09 # snapshot compressed 目錄，取個隨便名稱就用今天日期
md5 /pool_name/dataset_name/.zfs/snapshot/2009-04-09 # 再次用 hash 去驗證 snapshot 前後所得到的檔案是否一樣：MD5 (/pool_name/dataset_name/.zfs/snapshot/2009-04-09) = 7bffed2808dfba7915f89f8f42b09f83

在 zfs 硬碟上建立 容量為 8G 的 swap (儲存池名稱 pool_name)
zfs create -V 8G pool_name/swap
zfs set checksum=off pool_name/swap
zfs set dd:swap=on pool_name/swap # dd 是自行設定的名稱
