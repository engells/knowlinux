Commands
====================
nmcli general status
 # 查詢網路狀態
nmcli device or nmcli device status
 # 查詢所有網路裝置
nmcli connection or nmcli connection show
 # 查詢所有連線設定
nmcli -f name,uuid,type,device,state con
 # 查詢所有連線設定，顯示指定欄位
nmcli connection modify 'xxxx'
 # 修改連線屬性值
nmcli radio wifi [on|off]
 # 啟動/關閉 wifi
nmcli device wifi show-password
 # 查詢連線的 SSID 等相關資料




建立有線網路連線
====================
nmcli connection add type ethernet con-name xxxx ifname eeee \
 ipv4.addresses 192.168.1.100/24 ipv4.gateway 192.168.1.1 ipv4.dns 8.8.8.8 8.8.4
 # 建立有線網路連線，取名 xxxx
sudo nmcli connection up xxxx
 # 啟用網路連線 xxxx

建立 bridge 連線
====================
nmcli connection mod Wired_connection_1 con-name cnn_wire0
 # 將連線 Wired connection 1 更名為 eth0-con
nmcli connection add con-name cnn_br0 type bridge ifname lxcbr0 stp no
 # 建立 bridge 裝置名稱為 lxcbr0，對應連線設定檔為 cnn_br0
 # con-name: Sets the NetworkManager connection profile name
 # ifname: Sets the actual Linux network interface name
nmcli connection add con-name cnn_br0_enp[x]s[y] type bridge-slave ifname enp[x]s[y] master lxcbr0
 # 將 enp[x]s[y] 綁定成為 lxcbr0 的 slave，對應之新連線設定檔為 cnn_br0_enp[x]s[y]
nmcli connection modify lxcbr0 ipv4.addresses "192.168.xxx.xxx/24"
 # 分配固定 IP 給 lxcbr0
nmcli connection modify lxcbr0 ipv4.dns "8.8.8.8,8.8.4.4"
 # 設定 lxcbr0 使用 Google 的 DNS
nmcli connection down cnn_wire0 && nmcli connection modify cnn_wire0 connection.autoconnect no \
nmcli connection up cnn_br0 && nmcli connection up cnn_br0_enp[x]s[y]
 # 虛擬機提升至跟 host 相同的網段，採用 cnn_br0 連線
nmcli connection down cnn_br0 && nmcli connection down cnn_br0_enp[x]s[y] \
nmcli connection up cnn_wire0 && nmcli connection modify cnn_wire0 connection.autoconnect yes
 # 採用 cnn_wire0 直接連網
ip addr show lxcbr0
 # 確認連線狀態

物理接口 enp3s0 => VLAN 接口 vlan10 => 网桥 br0 => 虚拟机/容器

创建 VLAN 接口
nmcli connection add type vlancon-name vlan10 dev enp3s0 id 10 ifname vlan10

创建网桥接口
nmcli connection add type bridge con-name br0 ifname br0
nmcli connection modify br0 dridge.stp yes
nmcli connection modify br0 bridge.priority 4096

添加 VLAN10 作为网桥从属
nmcli connection add type bridge-slave con-name br-vlan10 ifname vlan10 master br0
bridge link show br0

配置 IP 地址
sudo nmcli connection modify br0 ipv4.addresses '192.168.10.100/24
sudo nmcli connection modify br0 ipv4.gateway '192.168.10.1'
sudo nmcli connection modify br0 ipv4.dns '8.8.8.8'
sudo nmcli connection modify br0 ipv4.method manual
sudo nmcli connection modify br0 ipv4.method auto

激活所有连接
sudo nmcli connection down "Wired connection 1"
sudo nmcli connection up vlan10
sudo nmcli connection up br-vlan10
sudo nmcli connection up br0
