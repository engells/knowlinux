lazyvim
====================
install
  mv ~/.config/nvim{,.bak}
  mv ~/.local/share/nvim{,.bak}
  mv ~/.local/state/nvim{,.bak}
  mv ~/.cache/nvim{,.bak}
  git clone https://github.com/LazyVim/starter ~/.config/nvim
  rm -rf ~/.config/nvim/.git
  nvim  # nvim will instal plugins

uninstall
  rm -rv $HOME/.local/share/nvim/           # remove lazyvim data
  rm -rv $HOME/.local/state/nvim/           # remove lazyvim state
  rm -rv $HOME/.config/nvim/lazy-lock.json  # remove lockfile
